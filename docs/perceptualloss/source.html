 3from torch import nn
 4from torchvision.models import vgg16

 6class VGG16PerceptualLoss(nn.Module):
 7    def __init__(self):
 8        """Initialize the VGG16 perceptual loss model.
 9        
10        It computes the perceptual loss as the mean squared error between the features.
11        
12        The model is set to evaluation mode and the parameters are frozen.
13        
14        **TODO**:
15
16        - Load the VGG16 model with pretrained weights. Use `torchvision.models.vgg16(pretrained=True)`.        
17
18        - Restrict the VGG16 model to the first 16 layers by using `self.vgg = vgg16(pretrained=True).features[:16]`.
19
20        - Set the model to evaluation mode using `.eval()`.
21
22        - Freeze the parameters of the VGG16 model by setting `param.requires_grad = False` for all parameters. 
23          NOTE: Iterate through all parameters by using the `self.vgg.parameters()`-Iterator.
24
25        - Initialize the L2 loss function using `nn.MSELoss()`.  
26        """
27        super(VGG16PerceptualLoss, self).__init__()
28        self.vgg = vgg16(pretrained=True).features[:16].eval().cuda()
29
30        for param in self.vgg.parameters():
31            param.requires_grad = False
32
33        self.l1_loss = nn.L1Loss()
34
35    def forward(self, output, target):
36        """Compute the perceptual loss between two images.
37        
38        Parameters:
39        -----------
40            output (torch.Tensor): 
41              The output image tensor from the upscaler network.
42
43            target (torch.Tensor): 
44              The target image tensor from ground truth.
45
46        Returns:
47        --------  
48            torch.Tensor: 
49              The computed perceptual loss as the mean squared error between the features of the two images.
50
51        **TODO**:
52
53        - Resize `output` and `target` to 224x224 using `torch.nn.functional.interpolate()`. Use `mode='bilinear'` and `align_corners=False`. 
54        
55        - Pass `output` through the VGG16 model to get the features `f1`.
56
57        - Pass `target` through the VGG16 model to get the features `f2`. Note: You should use `torch.no_grad()` to avoid computing gradients for the target image.
58
59        - Compute and return the L2 loss between `f1` and `f2` using `self.l2_loss(f1, f2)`.
60        """
61        output = torch.nn.functional.interpolate(output, size=(224, 224), mode='bilinear', align_corners=False)
62        target = torch.nn.functional.interpolate(target, size=(224, 224), mode='bilinear', align_corners=False)
63
64        f1 = self.vgg(output)
65
66        with torch.no_grad():
67          f2 = self.vgg(target)
68
69        return self.l1_loss(f1, f2)

 7class Upscale2x(nn.Module):
 8    def __init__(self):
 9        """Initialize the Upscale2x model.
10        
11        This model performs 2x upscaling using a series of ResNet blocks and an upsampling layer.
12        
13        **TODO**:
14
15        - Call the `__init__` method of the base class `nn.Module`.
16
17        - Define an upsampling layer using `nn.Upsample(scale_factor=2, mode="bilinear", align_corners=True) <https://docs.pytorch.org/docs/stable/generated/torch.nn.Upsample.html>`_.
18
19        - Define a sequential model consisting of:
20
21        - Three `ResNetBlock` layers with 3->64, 64->32 and 32->16 channels as well as kernel sizes 9, 7, and 5 respectively.
22
23        - A final convolutional layer with 16 input channels, 3 output channels and kernel size 5 with padding 2.
24        """
25        super(Upscale2x, self).__init__()
26        self.upsample = nn.Upsample(scale_factor=2, mode="bilinear", align_corners=True)
27        self.model = nn.Sequential(
28            ResNetBlock(3, 64, kernel_size=9),
29            ResNetBlock(64, 32, kernel_size=7),
30            ResNetBlock(32, 16, kernel_size=5),
31            nn.Conv2d(16, 3, kernel_size=5, padding=2),
32        )
33
34    def forward(self, x):
35        """Perform the forward pass of the Upscale2x model.
36
37        Parameters:
38        -----------
39            x (torch.Tensor): 
40              The input tensor to be upscaled.
41
42        Returns:
43        --------  
44            torch.Tensor: 
45              The upscaled output tensor. 
46
47        **TODO**: 
48
49        - Apply the upsampling layer to the input tensor `x`.
50
51        - Pass the upsampled tensor through the model.
52
53        - Add the upsampled tensor to the output of the model.  
54        """
55        up = self.upsample(x)
56        x = up + self.model(up)
57        return x
