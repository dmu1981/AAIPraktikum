 57        """

 59        self.conv1 = nn.Conv2d(3, 16, kernel_size=3, padding="same")
 60        self.conv2 = nn.Conv2d(16, 32, kernel_size=3, padding="same")
 61        self.conv3 = nn.Conv2d(32, 64, kernel_size=3, padding="same")

 63        self.bn1 = nn.BatchNorm2d(16)
 64        self.bn2 = nn.BatchNorm2d(32)

 70        self.pool = nn.MaxPool2d(kernel_size=2, stride=2)

 73    def forward(self, x):
 74        """Führt den Vorwärtsdurchlauf des Netzwerks aus.

 76        **TODO**:

 80        - Flatten Sie die Ausgabe der letzten Faltungs-Schicht mit .`self.flatten(x)`

 82        - Wenden Sie die voll verbundenen Schichten `fc1` und `fc2` auf die flachgelegte Ausgabe an, wobei Sie ReLU-Aktivierung auf die Ausgabe von `fc1` anwenden.

 84        - Geben Sie die Ausgabe der letzten Schicht `fc2` zurück.
 85        """

 90        x = torch.relu(self.fc1(x))
 91        x = self.dropout(x)

 95def epoch(model, n, train, dataloader, criterion, optimizer):
 96    """
 97    Führt eine einzelne Trainings- oder Evaluations-Epoche für das Modell aus.
 98
 99    Parameters:
100    -----------
101
102    model (nn.Module): 
103        Das zu trainierende oder evaluierende Modell.
104
105    n (int): 
106        Die aktuelle Epoche.   
107
108    train (bool):
109        Gibt an, ob die Epoche im Trainingsmodus oder im Evaluationsmodus durchgeführt wird.
110
111    dataloader (DataLoader):
112        Der DataLoader, der die Daten für die Epoche bereitstellt.
113
114    criterion (nn.Module):
115        Das Loss-Kriterium, das zur Berechnung des Verlusts verwendet wird.
116
117    optimizer (torch.optim.Optimizer):
118        Der Optimierer, der zur Aktualisierung der Modellparameter verwendet wird.
119
120    **TODO**:
121    
122    - Setzen Sie das Modell in den Trainingsmodus, wenn `train=True` ist, und in den Evaluationsmodus, wenn `train=False` ist. Rufen Sie dazu `model.train()` bzw. `model.eval()` auf. 
123    
124    - Initialisieren Sie `total_loss`, `total_samples` und `total_correct` auf 0.0, 0 und 0. 
125
126    - Verwenden Sie `tqdm` für den Fortschrittsbalken, um den Fortschritt der Epoche anzuzeigen. 
127      Speichern Sie den Iterator in einer eigenen Variable damit er innerhalb der Schleife verwendet werden kann.
128
129    - Iterieren Sie über den `dataloader` und führen Sie die folgenden Schritte aus:  
130
131    - Verschieben Sie die Daten und Labels auf das Gerät (`DEVICE`).
132
133    - Setzen Sie die Gradienten des Optimierers zurück, wenn `train=True` ist, indem Sie `optimizer.zero_grad()` aufrufen.
134
135    - Führen Sie den Vorwärtsdurchlauf des Modells aus, indem Sie `model(data)` aufrufen. Verwenden Sie `torch.set_grad_enabled(train)`, um den Gradientenfluss nur im Trainingsmodus zu aktivieren.
136    
137    - Berechnen Sie den Verlust mit `criterion(outputs, labels)`.
138
139    - Wenn `train=True` ist, führen Sie den Rückwärtsdurchlauf aus, indem Sie `loss.backward()` aufrufen und die Parameter mit `optimizer.step()` aktualisieren.
140
141    - Aktualisieren Sie `total_loss`, `total_samples` und `total_correct` mit den entsprechenden Werten aus dem aktuellen Batch.
142      Der `total_loss` sollte den Verlust des aktuellen Batches aufsummieren, `total_samples` die Anzahl der Samples im aktuellen Batch und 
143      `total_correct` die Anzahl der korrekt klassifizierten Samples. Die Anzahl der korrekt klassifizierten Samples kann mit `(outputs.argmax(dim=1) == labels).sum()` berechnet werden.
144
145    - Aktualisieren Sie den Fortschrittsbalken mit dem aktuellen Verlust und der Genauigkeit. Zeigen Sie auch an ob das Netz im Trainings- oder Validationsmodus betrieben wird. 
146      Rufen Sie dazu tqdm.set_description() auf und formatieren Sie die Ausgabe entsprechend.
147    """
148    # Vorbereiten des Modells für Training oder Evaluation
149    if train:
150        model.train()
151    else:
152        model.eval()
153
154    # Training des Modells
155    total_loss = 0.0
156    total_samples = 0
157    total_correct = 0
158    bar = tqdm(dataloader)
159    for data, labels in bar:
160        # Daten und Labels auf das Gerät verschieben
161        data, labels = data.to(DEVICE), labels.to(DEVICE)
162        
163        # Gradienten zurücksetzen
164        if train:
165            optimizer.zero_grad()
166
167        # Vorwärtsdurchlauf
168        with torch.set_grad_enabled(train):
169            outputs = model(data)
170
171        # Verlust berechnen und Rückwärtsdurchlauf
172        loss = criterion(outputs, labels)
173
174        # Gradienten berechnen
175        if train:
176            loss.backward()
177            optimizer.step()
178
179        # Aktualisieren der Metriken
180        total_loss += loss.item()
181        total_samples += data.size(0)
182        total_correct += (outputs.argmax(dim=1) == labels).sum().item()
183
184        bar.set_description(f"Epoch {n} ({'T' if train else 'V'}), Loss: {total_loss / total_samples:.4f}, Accuracy: {total_correct / total_samples:.2%}")
